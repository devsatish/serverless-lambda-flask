# Example of using Variadic Generics for more flexible type annotations
from typing import TypeVar, Generic

T = TypeVar('T')

class Container(Generic[T]):
    def __init__(self, *args: T) -> None:
        self.items = args

    def get_items(self) -> tuple[T, ...]:
        return self.items
